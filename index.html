Welcome to divTable presentation page

Base table:

Javascript:

baseTable = document.getElementById( 'baseTable' );
table = divTableMaker( baseTable, {
	columns : [
		{ name : 'text',           text : 'Text',            edit : 'text'                                },
		{ name : 'number',         text : 'Number',          edit : 'number'                              },
		{ name : 'select',         text : 'Select',          edit : [ '0', '1', '2' ]                     },
		{ name : 'advancedSelect', text : 'Advanced Select', edit : {1: 'first', 2: 'second', 3: 'third'} },
		{ name : 'checkbox',       text : 'Checkbox',        edit : 'checkbox'                            }
	]
});

CSS:

#baseTable {
	width  : 100%;
	height : 300px;
}

#baseTable .cell-0,
#baseTable .cell-1,
#baseTable .cell-2,
#baseTable .cell-3,
#baseTable .cell-4 {
	width : 20%;
}

Advanced edit table:

Javascript:

advTable  = document.getElementById( 'advTable'  );
table = divTableMaker( advTable, {
	columns : [
		{ name : 'text',           text : 'Text',            edit : 'text'                                },
		{ name : 'number',         text : 'Number',          edit : 'number'                              },
		{ name : 'select',         text : 'Select',          edit : [ '0', '1', '2' ]                     },
		{ name : 'advancedSelect', text : 'Advanced Select', edit : {1: 'first', 2: 'second', 3: 'third'} },
		{ name : 'checkbox',       text : 'Checkbox',        edit : 'checkbox'                            }
	],
	advancedEdit : {
		hide   : [],
		edit   : {},
		depend : 'select',
		values : [{
			value : '0',
			hide  : [ 'checkbox' ],
		},{
			value : '1',
			hide  : [ 'number', 'text' ],
			edit  : {
				'checkbox' : [ '1', '2', '3', '4', '5' ]
			}
		}]
	}
});

CSS:

#advTable {
	width  : 100%;
	height : 300px;
}

#advTable .cell-0,
#advTable .cell-1,
#advTable .cell-2,
#advTable .cell-3,
#advTable .cell-4 {
	width : 20%;
}
